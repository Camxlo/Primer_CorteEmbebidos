74void TaskReadTempHum(void *pvParameters) {
75 Serial.println("[TaskReadTempHum] Iniciada la tarea de lectura DHT.");
76 for (;;) {
77 Serial.println("[TaskReadTempHum] Leyendo DHT...");
78 float temp = dht.readTemperature();
79 float hum = dht.readHumidity();
80
81 if (isnan(temp) || isnan(hum)) {
82 Serial.println("[TaskReadTempHum] Error leyendo DHT (NaN).");
83 } else {
84 temperature = temp;
85 humidity = hum;
86 Serial.print("[TaskReadTempHum] Temp: ");
87 Serial.print(temperature);
88 Serial.print("°C, Hum: ");
89 Serial.print(humidity);
90 Serial.println("%");
91
92 // Condición de alarma: temperatura > 24°C y humedad > 70%
93 if (temperature > 24 && humidity > 70) {
94 alarmTriggered = true;
95 Serial.println("[TaskReadTempHum] Alarma activada (Temp/Hum).");
96 } else {
97 alarmTriggered = false;
98 }
99 }
100 vTaskDelay(pdMS_TO_TICKS(2500));
101 }
102}
